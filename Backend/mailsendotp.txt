const nodemailer =
require("nodemailer");

const sendEmail =
async(email,otp)=>{

const transporter =
nodemailer.createTransport({

service:"gmail",

auth:{

user:
process.env.EMAIL,

pass:
process.env.EMAIL_PASSWORD

}

});

await transporter.sendMail({

from:
process.env.EMAIL,

to:
email,

subject:
"Password Reset OTP",

text:
`Your OTP is ${otp}`

});

};

module.exports =
sendEmail;
